\textbf{Question28:}
\\
 The Fourier transform $X(\omega)$ of $x(t) = e^{-t^2}$ is\\
Note:$\int_{-\infty}^{\infty} e^{-y^2} \,dy = \sqrt{\pi}$ \\  
A) $\sqrt{\pi} e^{\frac{\omega^2}{2}}$ \\
B) $\frac{e^{\frac{-\omega^2}{4}}}{2\sqrt{\pi}}$ \\
C) $\sqrt{\pi} e^{\frac{-\omega^2}{4}}$ \\
D) $\sqrt{\pi} e^{\frac{-\omega^2}{2}}$\\
\hfill Gate 2023 EC Question 28\\
\textbf{Solution}
\begin{align}
x^{'}(t)=-2te^{-t^{2}}\\
x^{'}(t)=-2tx(t)
\end{align}
doing laplace transform 
\begin{align}
sX(s)= 2 \frac{d X(s)}{ds}\\
\int_{0}^{s} \frac{d X(s)}{X(s)}=\int_{0}^{s} \frac{sds}{2}\\
\frac{X(s)}{X(0)}=e^{\frac{s^2}{4}}\\
x(0)=\int_{-\infty}^{\infty} x(t)\,dy = \sqrt{\pi}\\
X(s)=\sqrt{\pi}e^{\frac{s^2}{4}}\\
s=j\omega\\
X(\omega)=\sqrt{\pi}e^{\frac{-\omega^2}{4}}
\end{align}
